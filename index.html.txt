/* ---------- AUTH ---------- */
let users=JSON.parse(localStorage.users||"{}")
async function hash(t){
  return [...new Uint8Array(await crypto.subtle.digest("SHA-256",new TextEncoder().encode(t)))]
  .map(b=>b.toString(16).padStart(2,"0")).join("")
}

async function signup(){
  if(users[user.value]) return msg.innerText="Exists"
  users[user.value]=await hash(pass.value)
  localStorage.users=JSON.stringify(users)
  msg.innerText="Signed up"
}
async function login(){
  if(users[user.value]!==await hash(pass.value)) return msg.innerText="Wrong"
  localStorage.current=user.value
  start()
}
function start(){
  auth.classList.add("hidden")
  app.classList.remove("hidden")
  pfp.innerText=localStorage.current[0].toUpperCase()
}
if(localStorage.current) start()
